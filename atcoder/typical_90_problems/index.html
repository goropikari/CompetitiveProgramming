001 - Yokan Party（★4）

問題

得られるスコアを二分探索で調べる. 長さ xxx 以上のピースに分割したとき, K+1K+1K+1 個以上ある場合は最低限スコア xxx は達成でき, 逆に K+1K+1K+1 個未満のときはスコア xxx は達成できない. (切りすぎたピースはくっつけて全体で K+1K+1K+1 個にすればよいので判定時には K+1K+1K+1 以上でよい)

判定するところを以下のように書いたら WA となった.

void solve() {
    ll n, l, k;
    cin >> n >> l >> k;
    vector<ll> a(n+2, 0);
    rep(i,n) cin >> a[i+1];
    a[n+1] = l;

    ll ans = 0;
    ll ok = 1, ng = l;
    while (abs(ok - ng) > 1) {
        ll mid = (ok + ng) / 2;
        ll len = 0;
        int cnt = 0;
        rep2(i,1,n+2) {
            len += a[i] - a[i-1];
            if (len >= mid) {
                cnt++;
                len = 0;
            }
        }

        if (cnt > k) {
            ok = mid;
            chmax(ans, ok);
        } else {
            ng = mid;
        }
    }
    cout << ans << endl;
}

chmax(ans, ok) で ans が更新されるからこれで良いと思っていたのだが N=1N = 1N=1, L=2L = 2L=2, K=1K = 1K=1, A1=1A_1 = 1A1​=1 のとき答えは 1 であるが, 上のプログラムだと 0 を返していた. このような条件のときに while の中にそもそも入らないのが原因だった.

そのため ans を使わない以下のように書けば AC になった.

ll ok = 1, ng = l;
while (abs(ok - ng) > 1) {
    ll mid = (ok + ng) / 2;
    ll len = 0;
    int cnt = 0;
    rep2(i,1,n+2) {
        len += a[i] - a[i-1];
        if (len >= mid) {
            cnt++;
            len = 0;
        }
    }

    if (cnt > k) {
        ok = mid;
    } else {
        ng = mid;
    }
}
cout << ok << endl;

003 - Longest Circular Road（★4）

問題

木なので最短距離は幅優先探索でも Dijkstra でもどっちでも大丈夫.

006 - Smallest Subsequence（★5）

問題

解説AC

void solve() {
    int N, K;
    cin >> N >> K;
    string S;
    cin >> S;

    vvi nex(N+1, vi(26, 0));
    rep(i,26) nex[N][i] = N; // 残りの文字がないことを表す番兵

    for (int i = N-1; i >= 0; i--) {
        for (int j = 0; j < 26; j++) {
            if (S[i] - 'a' == j) nex[i][j] = i;
            else nex[i][j] = nex[i+1][j];
        }
    }

    string ans = "";
    int cur = 0;
    for (int i = 1; i <= K; i++) {
        for (int j = 0; j < 26; j++) {
            int nextPos = nex[cur][j];
            int numRem = N-1 - nextPos; // nextPos 以降にある文字の数
            int needNum = K - i; // K 文字構成するのに必要な残りの数
            if (numRem >= needNum) {
                ans.push_back(S[nextPos]);
                cur = nextPos + 1;
                break;
            }
        }
    }

    cout << ans << endl;
}

dp[i][w]dp[i][w]dp[i][w]: iii 番目までの香辛料を使って重さが www であるときの価値の最大値とする. dp[0][0]=0dp[0][0] = 0dp[0][0]=0 とし, それ以外の値は −∞-\infty−∞ で初期化する.

iii 番目の香辛料を使って重さが www とするときの価値の最大値は dp[i][w]=max⁡j∈[L,R]{dp[i−1][w−j]}+V\displaystyle dp[i][w] = \max_{j \in [L,R]} \{ dp[i-1][w-j] \} + Vdp[i][w]=j∈[L,R]max​{dp[i−1][w−j]}+V となり, 使わない場合は dp[i][w]=dp[i−1][w]dp[i][w] = dp[i-1][w]dp[i][w]=dp[i−1][w] である.

これを実装すると以下のようになる.

vvll dp(N+1, vll(W+1, -INF));
dp[0][0] = 0;
rep2(i,1,N+1) {
    auto [l,r,v] = spices[i];
    rep2(w,0,W+1) {
        rep2(j,l,r+1) {
            if (w-j >= 0) {
                chmax(dp[i][w], dp[i-1][w-j] + v);
            }
        }
        chmax(dp[i][w], dp[i-1][w]);
    }
}

if (dp[N][W] < 0) cout << -1 << endl;

rep2(i,1,N+1) {

            int L = max(0ll, w-r);
            int R = w - l;
            ll mx = seg.prod(L, R+1);
            if (mx + v > seg.get(w)) seg.set(w, mx + v);
