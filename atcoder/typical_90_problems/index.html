001 - Yokan Party（★4）

問題

得られるスコアを二分探索で調べる. 長さ xxx 以上のピースに分割したとき, K+1K+1K+1 個以上ある場合は最低限スコア xxx は達成でき, 逆に K+1K+1K+1 個未満のときはスコア xxx は達成できない. (切りすぎたピースはくっつけて全体で K+1K+1K+1 個にすればよいので判定時には K+1K+1K+1 以上でよい)

判定するところを以下のように書いたら WA となった.

void solve() {
    ll n, l, k;
    cin >> n >> l >> k;
    vector<ll> a(n+2, 0);
    rep(i,n) cin >> a[i+1];
    a[n+1] = l;

    ll ans = 0;
    ll ok = 1, ng = l;
    while (abs(ok - ng) > 1) {
        ll mid = (ok + ng) / 2;
        ll len = 0;
        int cnt = 0;
        rep2(i,1,n+2) {
            len += a[i] - a[i-1];
            if (len >= mid) {
                cnt++;
                len = 0;
            }
        }

        if (cnt > k) {
            ok = mid;
            chmax(ans, ok);
        } else {
            ng = mid;
        }
    }
    cout << ans << endl;
}

chmax(ans, ok) で ans が更新されるからこれで良いと思っていたのだが N=1N = 1N=1, L=2L = 2L=2, K=1K = 1K=1, A1=1A_1 = 1A1​=1 のとき答えは 1 であるが, 上のプログラムだと 0 を返していた. このような条件のときに while の中にそもそも入らないのが原因だった.

そのため ans を使わない以下のように書けば AC になった.

}

006 - Smallest Subsequence（★5）

問題

解説AC

void solve() {
    int N, K;
    cin >> N >> K;
    string S;
    cin >> S;

    vvi nex(N+1, vi(26, 0));
    rep(i,26) nex[N][i] = N; // 残りの文字がないことを表す番兵

    for (int i = N-1; i >= 0; i--) {
        for (int j = 0; j < 26; j++) {
            if (S[i] - 'a' == j) nex[i][j] = i;
            else nex[i][j] = nex[i+1][j];
        }
    }

    string ans = "";
    int cur = 0;
    for (int i = 1; i <= K; i++) {
        for (int j = 0; j < 26; j++) {
            int nextPos = nex[cur][j];
            int numRem = N-1 - nextPos; // nextPos 以降にある文字の数
            int needNum = K - i; // K 文字構成するのに必要な残りの数
            if (numRem >= needNum) {
                ans.push_back(S[nextPos]);
                cur = nextPos + 1;
                break;
            }
        }
    }

    cout << ans << endl;
}

void solve() {
    int N, S;
    cin >> N >> S;
    vvint bag(N+1, vint(2));
    rep(i,N) rep(j,2) {
        cin >> bag[i+1][j];
    }

    // dp[i][x][S]: i 日目に福袋 x を買って総額が S となるか
    vector<vvint> dp(N+1, vvint(2, vint(S+1)));
    // 0 日目には何も買わないので総額が0円となることは可能
    dp[0][0][0] = dp[0][1][0] = 1;
    rep2(i,1,N+1) {
        rep(s,S+1) {
            rep(x,2) {
                // i 日目に x を買う
                rep(y,2) {
                    // i-1 日目に y を買う
                    if (s-bag[i][x] >= 0) dp[i][x][s] |= dp[i-1][y][s-bag[i][x]];
                }
            }
        }
    }

    if (dp[N][0][S] == 0 && dp[N][1][S] == 0) {
        cout << "Impossible" << endl;
        return;
    }

    int s = S;
    string ans = "";
    for (int i = N; i >= 1; i--) {
        rep(x,2) {
            if (dp[i][x][s]) {
                ans.push_back((x == 0 ? 'A' : 'B'));
                int t = s - bag[i][x];
                if (t >= 0) {
                    if (dp[i-1][0][t]) {
                        s = t;
                        break;
                    }
                    if (dp[i-1][1][t]) {
                        s = t;
                        break;
                    }
                }
            }
        }
    }
    reverse(all(ans));
    cout << ans << endl;
}

void solve() {
    int n, q;
    cin >> n >> q;
    dsu uf(n);
    fenwick_tree<ll> fw(n+1);

    rep(i,q) {
        ll t, x, y, v;
        cin >> t >> x >> y >> v;
        x--, y--;
        if (t == 0) {
            if (!uf.same(x,y)) {
                if (x % 2 == 0) fw.add(x, v);
                else fw.add(x, -v);
                uf.merge(x,y);
            }
        } else {
            if (!uf.same(x,y)) {
                cout << "Ambiguous" << endl;
                continue;
            }
            ll ox = x, oy = y;
            if (x > y) swap(x,y);
            ll sum = fw.sum(x, y);
            if (x % 2 != y % 2) {
                if (x % 2 == 1) sum *= -1;
                cout << sum - v << endl;
            } else {
                if (x % 2 == 0) {
                    // ax - ay
                    if (ox < oy) {
                        sum *= -1ll;
                    }
                } else {
                    // -ax + ay
                    if (ox > oy) {
                        sum *= -1ll;
                    }
                }
                cout << sum + v << endl;
            }
        }
    }
}

mint sum(ll l, ll r) {
    if (l >= r) return 0;

    mint L = l, R = r;
    mint ans = R * (R-1) / 2;
    ans -= (L-1)*L / 2;

        ll r = min(R+1, base*10);
        ans += sum(l, r) * digit;
        base *= 10;
        digit++;
    }

    if (R == (ll)1e18) ans += mint(ll(1e18)) * 19;

    cout << ans.val() << endl;
