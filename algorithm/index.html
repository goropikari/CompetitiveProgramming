priority queue

昇順

#include <iostream>
#include <queue>

int main()
{
  // intを要素に持つ優先順位付きキュー。
  // 降順に処理する
  std::priority_queue<int> que;

  // データを追加する
  que.push(3);
  que.push(1);
  que.push(4);

  // 処理順に出力する
  while (!que.empty()) {
    std::cout << que.top() << std::endl;
    que.pop();
  }
}

sort

自作 struct に関して sort する。

#include <iostream>
#include <vector>
#include <algorithm>

using namespace std;

struct Hoge {
    int x, y;
};

bool compare(Hoge a, Hoge b) { return a.y > b.y; }

int main(int argc, char* argv[])
{
    vector<Hoge> v = { {0, 1}, {1, 9} };

    sort(v.begin(), v.end(), compare);

    for (Hoge item : v) {
        cout << item.x << " " << item.y << endl;
    }
    return 0;
}

int gcd(int a, int b) {
    if (a == 0) return b;
    return gcd(b, a%b);
}

int lcm(int a, int b) {
    return (a*b) / gcd(a,b);
}

cumsum

template<typename T>
vector<T> cumsum(vector<T> v) {
    int n = v.size();
    rep2(i,1,n) v[i] += v[i-1];
    return v;
}
