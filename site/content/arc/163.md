## B. Favorite Game

<https://atcoder.jp/contests/arc163/tasks/arc163_b>

自力 AC

```cpp
void solve() {
    ios_base::sync_with_stdio(false);
    cin.tie(nullptr);

    int N, M;
    cin >> N >> M;
    vll A(N);
    rep(i, N) cin >> A[i];

    ll l = A[0], r = A[1];
    vll B;
    rep2(i, 2, N) B.push_back(A[i]);
    sort(all(B));

    ll ans = INF;
    for (int i = 0; i + M - 1 < N - 2; i++) {
        ll left = B[i], right = B[i + M - 1];
        ll tmp = 0;
        if (left < l) tmp += l - left;
        if (r < right) tmp += right - r;
        chmin(ans, tmp);
    }
    cout << ans << endl;
}
```
